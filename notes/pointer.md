#### complex pointer

```c
int board[8][8];    // 声明一个内含int数组的数组
int ** ptr;         // 声明一个指向指针的指针，被指向的指针指向int
int * risks[10];    // 声明一个内含10个元素的数组，每个元素都是一个指向int的指针
int (*risks)[10];   // 声明一个指向数组的指针，该数组内含10个int类型的值
int * oof[3][4];    // 声明一个 3*4 的二维数组，每个元素都是指向int的指针
int (* uuf)[3][4];  // 声明一个指向 3*4 的二维数组的指针，该数组内含int类型的值
int (* uuf[3])[4];  // 声明一个内含3个指针元素的数组，其中每个指针都指向一个内含4个int类型元素的数组

```
